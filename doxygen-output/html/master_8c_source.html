677 int ec_master_enter_operation_phase(
678  ec_master_t *master
679  )
680 {
681  int ret = 0;
682  ec_slave_t *slave;
683 #ifdef EC_EOE
684  ec_eoe_t *eoe;
685 #endif
686 
687  EC_MASTER_DBG(master, 1, "IDLE -> OPERATION.\n");
688 
689  down(&master->config_sem);
690  if (master->config_busy) {
691  up(&master->config_sem);
692 
693  // wait for slave configuration to complete
694  ret = wait_event_interruptible(master->config_queue,
695  !master->config_busy);
696  if (ret) {
697  EC_MASTER_INFO(master, "Finishing slave configuration"
698  " interrupted by signal.\n");
699  goto out_return;
700  }
701 
702  EC_MASTER_DBG(master, 1, "Waiting for pending slave"
703  " configuration returned.\n");
704  } else {
705  up(&master->config_sem);
706  }
707 
708  down(&master->scan_sem);
709  master->allow_scan = 0; // 'lock' the slave list
710  if (!master->scan_busy) {
711  up(&master->scan_sem);
712  } else {
713  up(&master->scan_sem);
714 
715  // wait for slave scan to complete
716  ret = wait_event_interruptible(master->scan_queue,
717  !master->scan_busy);
718  if (ret) {
719  EC_MASTER_INFO(master, "Waiting for slave scan"
720  " interrupted by signal.\n");
721  goto out_allow;
722  }
723 
724  EC_MASTER_DBG(master, 1, "Waiting for pending"
725  " slave scan returned.\n");
726  }
727 
728  // set states for all slaves
729  for (slave = master->slaves;
730  slave < master->slaves + master->slave_count;
731  slave++) {
732  ec_slave_request_state(slave, EC_SLAVE_STATE_PREOP);
733  }
734 
735 #ifdef EC_EOE
736  // ... but set EoE slaves to OP
737  list_for_each_entry(eoe, &master->eoe_handlers, list) {
738  if (ec_eoe_is_open(eoe))
739  ec_slave_request_state(eoe->slave, EC_SLAVE_STATE_OP);
740  }
741 #endif
742 
743  master->phase = EC_OPERATION;
744  master->app_send_cb = NULL;
745  master->app_receive_cb = NULL;
746  master->app_cb_data = NULL;
747  return ret;
748 
749 out_allow:
750  master->allow_scan = 1;
751 out_return:
752  return ret;
753 }

972 void ec_master_send_datagrams(
973  ec_master_t *master,
974  ec_device_index_t device_index
975  )
976 {
977  ec_datagram_t *datagram, *next;
978  size_t datagram_size;
979  uint8_t *frame_data, *cur_data = NULL;
980  void *follows_word;
981 #ifdef EC_HAVE_CYCLES
982  cycles_t cycles_start, cycles_sent, cycles_end;
983 #endif
984  unsigned long jiffies_sent;
985  unsigned int frame_count, more_datagrams_waiting;
986  struct list_head sent_datagrams;
987 
988 #ifdef EC_HAVE_CYCLES
989  cycles_start = get_cycles();
990 #endif
991  frame_count = 0;
992  INIT_LIST_HEAD(&sent_datagrams);
993 
994  EC_MASTER_DBG(master, 2, "%s(device_index = %u)\n",
995  __func__, device_index);
996 
997  do {
998  frame_data = NULL;
999  follows_word = NULL;
1000  more_datagrams_waiting = 0;
1001 
1002  // fill current frame with datagrams
1003  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1004  if (datagram->state != EC_DATAGRAM_QUEUED ||
1005  datagram->device_index != device_index) {
1006  continue;
1007  }
1008 
1009  if (!frame_data) {
1010  // fetch pointer to transmit socket buffer
1011  frame_data =
1012  ec_device_tx_data(&master->devices[device_index]);
1013  cur_data = frame_data + EC_FRAME_HEADER_SIZE;
1014  }
1015 
1016  // does the current datagram fit in the frame?
1017  datagram_size = EC_DATAGRAM_HEADER_SIZE + datagram->data_size
1018  + EC_DATAGRAM_FOOTER_SIZE;
1019  if (cur_data - frame_data + datagram_size > ETH_DATA_LEN) {
1020  more_datagrams_waiting = 1;
1021  break;
1022  }
1023 
1024  list_add_tail(&datagram->sent, &sent_datagrams);
1025  datagram->index = master->datagram_index++;
1026 
1027  EC_MASTER_DBG(master, 2, "Adding datagram 0x%02X\n",
1028  datagram->index);
1029 
1030  // set "datagram following" flag in previous datagram
1031  if (follows_word) {
1032  EC_WRITE_U16(follows_word,
1033  EC_READ_U16(follows_word) | 0x8000);
1034  }
1035 
1036  // EtherCAT datagram header
1037  EC_WRITE_U8 (cur_data, datagram->type);
1038  EC_WRITE_U8 (cur_data + 1, datagram->index);
1039  memcpy(cur_data + 2, datagram->address, EC_ADDR_LEN);
1040  EC_WRITE_U16(cur_data + 6, datagram->data_size & 0x7FF);
1041  EC_WRITE_U16(cur_data + 8, 0x0000);
1042  follows_word = cur_data + 6;
1043  cur_data += EC_DATAGRAM_HEADER_SIZE;
1044 
1045  // EtherCAT datagram data
1046  memcpy(cur_data, datagram->data, datagram->data_size);
1047  cur_data += datagram->data_size;
1048 
1049  // EtherCAT datagram footer
1050  EC_WRITE_U16(cur_data, 0x0000); // reset working counter
1051  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1052  }
1053 
1054  if (list_empty(&sent_datagrams)) {
1055  EC_MASTER_DBG(master, 2, "nothing to send.\n");
1056  break;
1057  }
1058 
1059  // EtherCAT frame header
1060  EC_WRITE_U16(frame_data, ((cur_data - frame_data
1061  - EC_FRAME_HEADER_SIZE) & 0x7FF) | 0x1000);
1062 
1063  // pad frame
1064  while (cur_data - frame_data < ETH_ZLEN - ETH_HLEN)
1065  EC_WRITE_U8(cur_data++, 0x00);
1066 
1067  EC_MASTER_DBG(master, 2, "frame size: %zu\n", cur_data - frame_data);
1068 
1069  // send frame
1070  ec_device_send(&master->devices[device_index],
1071  cur_data - frame_data);
1072 #ifdef EC_HAVE_CYCLES
1073  cycles_sent = get_cycles();
1074 #endif
1075  jiffies_sent = jiffies;
1076 
1077  // set datagram states and sending timestamps
1078  list_for_each_entry_safe(datagram, next, &sent_datagrams, sent) {
1079  datagram->state = EC_DATAGRAM_SENT;
1080 #ifdef EC_HAVE_CYCLES
1081  datagram->cycles_sent = cycles_sent;
1082 #endif
1083  datagram->jiffies_sent = jiffies_sent;
1084  list_del_init(&datagram->sent); // empty list of sent datagrams
1085  }
1086 
1087  frame_count++;
1088  }
1089  while (more_datagrams_waiting);
1090 
1091 #ifdef EC_HAVE_CYCLES
1092  if (unlikely(master->debug_level > 1)) {
1093  cycles_end = get_cycles();
1094  EC_MASTER_DBG(master, 0, "%s()"
1095  " sent %u frames in %uus.\n", __func__, frame_count,
1096  (unsigned int) (cycles_end - cycles_start) * 1000 / cpu_khz);
1097  }
1098 #endif
1099 }

1109 void ec_master_receive_datagrams(ec_master_t *master,
1110  const uint8_t *frame_data,
1111  size_t size
1112  )
1113 {

1115  uint8_t datagram_type, datagram_index;
1116  unsigned int cmd_follows, matched;
1117  const uint8_t *cur_data;
1118  ec_datagram_t *datagram;
1119 
1120  if (unlikely(size < EC_FRAME_HEADER_SIZE)) {
1121  if (master->debug_level) {
1122  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1123  " (size %zu < %u byte):\n",
1124  size, EC_FRAME_HEADER_SIZE);
1125  ec_print_data(frame_data, size);
1126  }
1127  master->stats.corrupted++;
1128  ec_master_output_stats(master);
1129  return;
1130  }
1131 
1132  cur_data = frame_data;
1133 
1134  // check length of entire frame
1135  frame_size = EC_READ_U16(cur_data) & 0x07FF;
1136  cur_data += EC_FRAME_HEADER_SIZE;
1137 
1138  if (unlikely(frame_size > size)) {
1139  if (master->debug_level) {
1140  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1141  " (invalid frame size %zu for "
1142  "received size %zu):\n", frame_size, size);
1143  ec_print_data(frame_data, size);
1144  }
1145  master->stats.corrupted++;
1146  ec_master_output_stats(master);
1147  return;
1148  }
1149 
1150  cmd_follows = 1;
1151  while (cmd_follows) {
1152  // process datagram header
1153  datagram_type = EC_READ_U8 (cur_data);
1154  datagram_index = EC_READ_U8 (cur_data + 1);
1155  data_size = EC_READ_U16(cur_data + 6) & 0x07FF;
1156  cmd_follows = EC_READ_U16(cur_data + 6) & 0x8000;
1157  cur_data += EC_DATAGRAM_HEADER_SIZE;
1158 
1159  if (unlikely(cur_data - frame_data
1160  + data_size + EC_DATAGRAM_FOOTER_SIZE > size)) {
1161  if (master->debug_level) {
1162  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1163  " (invalid data size %zu):\n", data_size);
1164  ec_print_data(frame_data, size);
1165  }
1166  master->stats.corrupted++;
1167  ec_master_output_stats(master);
1168  return;
1169  }
1170 
1171  // search for matching datagram in the queue
1172  matched = 0;
1173  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1174  if (datagram->index == datagram_index
1175  && datagram->state == EC_DATAGRAM_SENT
1176  && datagram->type == datagram_type
1177  && datagram->data_size == data_size) {
1178  matched = 1;
1179  break;
1180  }
1181  }
1182 
1183  // no matching datagram was found
1184  if (!matched) {
1185  master->stats.unmatched++;
1186  ec_master_output_stats(master);
1187 
1188  if (unlikely(master->debug_level > 0)) {
1189  EC_MASTER_DBG(master, 0, "UNMATCHED datagram:\n");
1190  ec_print_data(cur_data - EC_DATAGRAM_HEADER_SIZE,
1191  EC_DATAGRAM_HEADER_SIZE + data_size
1192  + EC_DATAGRAM_FOOTER_SIZE);
1193 #ifdef EC_DEBUG_RING
1194  ec_device_debug_ring_print(&master->devices[EC_DEVICE_MAIN]);
1195 #endif
1196  }
1197 
1198  cur_data += data_size + EC_DATAGRAM_FOOTER_SIZE;
1199  continue;
1200  }
1201 
1202  if (datagram->type != EC_DATAGRAM_APWR &&
1203  datagram->type != EC_DATAGRAM_FPWR &&
1204  datagram->type != EC_DATAGRAM_BWR &&
1205  datagram->type != EC_DATAGRAM_LWR) {
1206  // copy received data into the datagram memory,
1207  // if something has been read
1208  memcpy(datagram->data, cur_data, data_size);
1209  }
1210  cur_data += data_size;
1211 
1212  // set the datagram's working counter
1213  datagram->working_counter = EC_READ_U16(cur_data);
1214  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1215 
1216  // dequeue the received datagram
1217  datagram->state = EC_DATAGRAM_RECEIVED;
1218 #ifdef EC_HAVE_CYCLES
1219  datagram->cycles_received =
1220  master->devices[EC_DEVICE_MAIN].cycles_poll;
1221 #endif
1222  datagram->jiffies_received =
1223  master->devices[EC_DEVICE_MAIN].jiffies_poll;
1224  list_del_init(&datagram->queue);
1225  }
1226 }

1299  u64 loss;
