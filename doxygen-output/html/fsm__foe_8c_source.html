129 int ec_fsm_foe_exec(
130  ec_fsm_foe_t *fsm,
131  ec_datagram_t *datagram
132  )
133 {
134  int datagram_used = 0;
135 
136  if (fsm->datagram &&
137  (fsm->datagram->state == EC_DATAGRAM_INIT ||
138  fsm->datagram->state == EC_DATAGRAM_QUEUED ||
139  fsm->datagram->state == EC_DATAGRAM_SENT)) {
140  // datagram not received yet
141  return datagram_used;
142  }
143 
144  fsm->state(fsm, datagram);
145 
146  datagram_used =
147  fsm->state != ec_fsm_foe_end && fsm->state != ec_fsm_foe_error;
148 
149  if (datagram_used) {
150  fsm->datagram = datagram;
151  } else {
152  fsm->datagram = NULL;
153  }
154 
155  return datagram_used;
156 }

236 int ec_foe_prepare_data_send(
237  ec_fsm_foe_t *fsm,
238  ec_datagram_t *datagram
239  )
240 {
241  size_t remaining_size, current_size;
242  uint8_t *data;
243 
244  remaining_size = fsm->tx_buffer_size - fsm->tx_buffer_offset;
245 
246  if (remaining_size < fsm->slave->configured_tx_mailbox_size
247  - EC_MBOX_HEADER_SIZE - EC_FOE_HEADER_SIZE) {
248  current_size = remaining_size;
249  fsm->tx_last_packet = 1;
250  } else {
251  current_size = fsm->slave->configured_tx_mailbox_size
252  - EC_MBOX_HEADER_SIZE - EC_FOE_HEADER_SIZE;
253  }
254 
255  data = ec_slave_mbox_prepare_send(fsm->slave,
256  datagram, EC_MBOX_TYPE_FILEACCESS,
257  current_size + EC_FOE_HEADER_SIZE);
258  if (IS_ERR(data)) {
259  return -1;
260  }
261 
262  EC_WRITE_U16(data, EC_FOE_OPCODE_DATA); // OpCode = DataBlock req.
263  EC_WRITE_U32(data + 2, fsm->tx_packet_no); // PacketNo, Password
264 
265  memcpy(data + EC_FOE_HEADER_SIZE,
266  fsm->tx_buffer + fsm->tx_buffer_offset, current_size);
267  fsm->tx_current_size = current_size;
268 
269  return 0;
270 }

345 void ec_fsm_foe_state_ack_check(
346  ec_fsm_foe_t *fsm,
347  ec_datagram_t *datagram
348  )
349 {
350  ec_slave_t *slave = fsm->slave;
351 
352 #ifdef DEBUG_FOE
353  printk("ec_fsm_foe_ack_check()\n");
354 #endif
355 
356  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
357  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
358  EC_SLAVE_ERR(slave, "Failed to receive FoE mailbox check datagram: ");
359  ec_datagram_print_state(fsm->datagram);
360  return;
361  }
362 
363  if (fsm->datagram->working_counter != 1) {
364  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
365  EC_SLAVE_ERR(slave, "Reception of FoE mailbox check datagram"
366  " failed: ");
367  ec_datagram_print_wc_error(fsm->datagram);
368  return;
369  }
370 
371  if (!ec_slave_mbox_check(fsm->datagram)) {
372  // slave did not put anything in the mailbox yet
373  unsigned long diff_ms =
374  (datagram->jiffies_received - fsm->jiffies_start) * 1000 / HZ;
375  if (diff_ms >= EC_FSM_FOE_TIMEOUT) {
376  ec_foe_set_tx_error(fsm, FOE_TIMEOUT_ERROR);
377  EC_SLAVE_ERR(slave, "Timeout while waiting for ack response.\n");
378  return;
379  }
380 
381  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
382  fsm->retries = EC_FSM_RETRIES;
383  return;
384  }
385 
386  // Fetch response
387  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
388 
389  fsm->retries = EC_FSM_RETRIES;
390  fsm->state = ec_fsm_foe_state_ack_read;
391 }

397 void ec_fsm_foe_state_ack_read(
398  ec_fsm_foe_t *fsm,
399  ec_datagram_t *datagram
400  )
401 {
402  ec_slave_t *slave = fsm->slave;
403  uint8_t *data, mbox_prot;
404  uint8_t opCode;
405  size_t rec_size;
406 
407 #ifdef DEBUG_FOE
408  printk("ec_fsm_foe_ack_read()\n");
409 #endif
410 
411  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
412  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
413  EC_SLAVE_ERR(slave, "Failed to receive FoE ack response datagram: ");
414  ec_datagram_print_state(fsm->datagram);
415  return;
416  }
417 
418  if (fsm->datagram->working_counter != 1) {
419  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
420  EC_SLAVE_ERR(slave, "Reception of FoE ack response failed: ");
421  ec_datagram_print_wc_error(fsm->datagram);
422  return;
423  }
424 
425  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
426  if (IS_ERR(data)) {
427  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
428  return;
429  }
430 
431  if (mbox_prot != EC_MBOX_TYPE_FILEACCESS) { // FoE
432  ec_foe_set_tx_error(fsm, FOE_MBOX_PROT_ERROR);
433  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
434  mbox_prot);
435  return;
436  }
437 
438  opCode = EC_READ_U8(data);
439 
440  if (opCode == EC_FOE_OPCODE_BUSY) {
441  // slave not ready
442  if (ec_foe_prepare_data_send(fsm, datagram)) {
443  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
444  EC_SLAVE_ERR(slave, "Slave is busy.\n");
445  return;
446  }
447  fsm->state = ec_fsm_foe_state_data_sent;
448  return;
449  }
450 
451  if (opCode == EC_FOE_OPCODE_ACK) {
452  fsm->tx_packet_no++;
453  fsm->tx_buffer_offset += fsm->tx_current_size;
454 
455  if (fsm->tx_last_packet) {
456  fsm->state = ec_fsm_foe_end;
457  return;
458  }
459 
460  if (ec_foe_prepare_data_send(fsm, datagram)) {
461  ec_foe_set_tx_error(fsm, FOE_PROT_ERROR);
462  return;
463  }
464  fsm->state = ec_fsm_foe_state_data_sent;
465  return;
466  }
467  ec_foe_set_tx_error(fsm, FOE_ACK_ERROR);
468 }

716  ec_foe_set_tx_error(fsm, FOE_TIMEOUT_ERROR);
717  EC_SLAVE_ERR(slave, "Timeout while waiting for ack response.\n");
718  return;
719  }
720 
721  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
722  fsm->retries = EC_FSM_RETRIES;
723  return;
724  }
725 
726  // Fetch response
727  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
728 
729  fsm->retries = EC_FSM_RETRIES;
730  fsm->state = ec_fsm_foe_state_data_read;
731 
732 }

738 void ec_fsm_foe_state_data_read(
739  ec_fsm_foe_t *fsm,
740  ec_datagram_t *datagram
741  )
742 {
743  size_t rec_size;
744  uint8_t *data, opCode, packet_no, mbox_prot;
745 
746  ec_slave_t *slave = fsm->slave;
747 
748 #ifdef DEBUG_FOE
749  printk("ec_fsm_foe_state_data_read()\n");
750 #endif
751 
752  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
753  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
754  EC_SLAVE_ERR(slave, "Failed to receive FoE DATA READ datagram: ");
755  ec_datagram_print_state(fsm->datagram);
756  return;
757  }
758 
759  if (fsm->datagram->working_counter != 1) {
760  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
761  EC_SLAVE_ERR(slave, "Reception of FoE DATA READ failed: ");
762  ec_datagram_print_wc_error(fsm->datagram);
763  return;
764  }
765 
766  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
767  if (IS_ERR(data)) {
768  ec_foe_set_rx_error(fsm, FOE_MBOX_FETCH_ERROR);
769  return;
770  }
771 
772  if (mbox_prot != EC_MBOX_TYPE_FILEACCESS) { // FoE
773  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
774  mbox_prot);
775  ec_foe_set_rx_error(fsm, FOE_PROT_ERROR);
776  return;
777  }
778 
779  opCode = EC_READ_U8(data);
780 
781  if (opCode == EC_FOE_OPCODE_BUSY) {
782  if (ec_foe_prepare_send_ack(fsm, datagram)) {
783  ec_foe_set_rx_error(fsm, FOE_PROT_ERROR);
784  }
785  return;
786  }
787 
788  if (opCode == EC_FOE_OPCODE_ERR) {
789  fsm->request->error_code = EC_READ_U32(data + 2);
790  EC_SLAVE_ERR(slave, "Received FoE Error Request (code 0x%08x).\n",
791  fsm->request->error_code);
792  if (rec_size > 6) {
793  uint8_t text[256];
794  strncpy(text, data + 6, min(rec_size - 6, sizeof(text)));
795  EC_SLAVE_ERR(slave, "FoE Error Text: %s\n", text);
796  }
797  ec_foe_set_rx_error(fsm, FOE_OPCODE_ERROR);
798  return;
799  }
800 
801  if (opCode != EC_FOE_OPCODE_DATA) {
802  EC_SLAVE_ERR(slave, "Received OPCODE %x, expected %x.\n",
803  opCode, EC_FOE_OPCODE_DATA);
804  fsm->request->error_code = 0x00000000;
805  ec_foe_set_rx_error(fsm, FOE_OPCODE_ERROR);
806  return;
807  }
808 
809  packet_no = EC_READ_U16(data + 2);
810  if (packet_no != fsm->rx_expected_packet_no) {
811  EC_SLAVE_ERR(slave, "Received unexpected packet number.\n");
812  ec_foe_set_rx_error(fsm, FOE_PACKETNO_ERROR);
813  return;
814  }
815 
816  rec_size -= EC_FOE_HEADER_SIZE;
817 
818  if (fsm->rx_buffer_size >= fsm->rx_buffer_offset + rec_size) {
819  memcpy(fsm->rx_buffer + fsm->rx_buffer_offset,
820  data + EC_FOE_HEADER_SIZE, rec_size);
821  fsm->rx_buffer_offset += rec_size;
822  }
823 
824  fsm->rx_last_packet =
825  (rec_size + EC_MBOX_HEADER_SIZE + EC_FOE_HEADER_SIZE
826  != slave->configured_rx_mailbox_size);
827 
828  if (fsm->rx_last_packet ||
829  (slave->configured_rx_mailbox_size - EC_MBOX_HEADER_SIZE
830  - EC_FOE_HEADER_SIZE + fsm->rx_buffer_offset)
831  <= fsm->rx_buffer_size) {
832  // either it was the last packet or a new packet will fit into the
833  // delivered buffer
834 #ifdef DEBUG_FOE
835  printk ("last_packet=true\n");
836 #endif
837  if (ec_foe_prepare_send_ack(fsm, datagram)) {
838  ec_foe_set_rx_error(fsm, FOE_RX_DATA_ACK_ERROR);
839  return;
840  }
841 
842  fsm->state = ec_fsm_foe_state_sent_ack;
843  }
844  else {
845  // no more data fits into the delivered buffer
846  // ... wait for new read request
847  printk ("ERROR: data doesn't fit in receive buffer\n");
848  printk (" rx_buffer_size = %d\n", fsm->rx_buffer_size);
849  printk (" rx_buffer_offset= %d\n", fsm->rx_buffer_offset);
850  printk (" rec_size = %zd\n", rec_size);
851  printk (" rx_mailbox_size = %d\n",
852  slave->configured_rx_mailbox_size);
853  printk (" rx_last_packet = %d\n", fsm->rx_last_packet);
854  fsm->request->result = FOE_READY;
855  }
856 }

862 void ec_fsm_foe_state_sent_ack(
863  ec_fsm_foe_t *fsm,
864  ec_datagram_t *datagram
865  )
866 {
867  ec_slave_t *slave = fsm->slave;
868 
869 #ifdef DEBUG_FOE
870  printk("ec_foe_state_sent_ack()\n");
871 #endif
872 
873  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
874  ec_foe_set_rx_error(fsm, FOE_RECEIVE_ERROR);
875  EC_SLAVE_ERR(slave, "Failed to send FoE ACK: ");
876  ec_datagram_print_state(fsm->datagram);
877  return;
878  }
879 
880  if (fsm->datagram->working_counter != 1) {
881  // slave did not put anything into the mailbox yet
882  ec_foe_set_rx_error(fsm, FOE_WC_ERROR);
883  EC_SLAVE_ERR(slave, "Reception of FoE ACK failed: ");
884  ec_datagram_print_wc_error(fsm->datagram);
885  return;
886  }
887 
888  fsm->jiffies_start = fsm->datagram->jiffies_sent;
889 
890  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
891 
892  if (fsm->rx_last_packet) {
893  fsm->rx_expected_packet_no = 0;
894  fsm->request->data_size = fsm->rx_buffer_offset;
895  fsm->state = ec_fsm_foe_end;
896  }
897  else {
898  fsm->rx_expected_packet_no++;
899  fsm->retries = EC_FSM_RETRIES;
900  fsm->state = ec_fsm_foe_state_data_check;
901  }
902 }
