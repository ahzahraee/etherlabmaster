214 void ec_fsm_master_state_broadcast(
215  ec_fsm_master_t *fsm
216  )
217 {
218  ec_datagram_t *datagram = fsm->datagram;
219  unsigned int i, size;
220  ec_slave_t *slave;
221  ec_master_t *master = fsm->master;
222 
223  // bus topology change?
224  if (datagram->working_counter != fsm->slaves_responding[fsm->dev_idx]) {
225  fsm->rescan_required = 1;
226  fsm->slaves_responding[fsm->dev_idx] = datagram->working_counter;
227  EC_MASTER_INFO(master, "%u slave(s) responding on %s device.\n",
228  fsm->slaves_responding[fsm->dev_idx],
229  ec_device_names[fsm->dev_idx != 0]);
230  }
231 
232  if (fsm->link_state[fsm->dev_idx] &&
233  !master->devices[fsm->dev_idx].link_state) {
234  ec_device_index_t dev_idx;
235 
236  EC_MASTER_DBG(master, 1, "Master state machine detected "
237  "link down on %s device. Clearing slave list.\n",
238  ec_device_names[fsm->dev_idx != 0]);
239 
240 #ifdef EC_EOE
241  ec_master_eoe_stop(master);
242  ec_master_clear_eoe_handlers(master);
243 #endif
244  ec_master_clear_slaves(master);
245 
246  for (dev_idx = EC_DEVICE_MAIN;
247  dev_idx < ec_master_num_devices(master); dev_idx++) {
248  fsm->slave_states[dev_idx] = 0x00;
249  fsm->slaves_responding[dev_idx] = 0; /* Reset to trigger rescan on
250  next link up. */
251  }
252  }
253  fsm->link_state[fsm->dev_idx] = master->devices[fsm->dev_idx].link_state;
254 
255  if (datagram->state == EC_DATAGRAM_RECEIVED &&
256  fsm->slaves_responding[fsm->dev_idx]) {
257  uint8_t states = EC_READ_U8(datagram->data);
258  if (states != fsm->slave_states[fsm->dev_idx]) {
259  // slave states changed
260  char state_str[EC_STATE_STRING_SIZE];
261  fsm->slave_states[fsm->dev_idx] = states;
262  ec_state_string(states, state_str, 1);
263  EC_MASTER_INFO(master, "Slave states on %s device: %s.\n",
264  ec_device_names[fsm->dev_idx != 0], state_str);
265  }
266  } else {
267  fsm->slave_states[fsm->dev_idx] = 0x00;
268  }
269 
270  fsm->dev_idx++;
271  if (fsm->dev_idx < ec_master_num_devices(master)) {
272  // check number of responding slaves on next device
273  fsm->state = ec_fsm_master_state_start;
274  fsm->state(fsm); // execute immediately
275  return;
276  }
277 
278  if (fsm->rescan_required) {
279  down(&master->scan_sem);
280  if (!master->allow_scan) {
281  up(&master->scan_sem);
282  } else {
283  unsigned int count = 0, next_dev_slave, ring_position;
284  ec_device_index_t dev_idx;
285 
286  master->scan_busy = 1;
287  up(&master->scan_sem);
288 
289  // clear all slaves and scan the bus
290  fsm->rescan_required = 0;
291  fsm->idle = 0;
292  fsm->scan_jiffies = jiffies;
293 
294 #ifdef EC_EOE
295  ec_master_eoe_stop(master);
296  ec_master_clear_eoe_handlers(master);
297 #endif
298  ec_master_clear_slaves(master);
299 
300  for (dev_idx = EC_DEVICE_MAIN;
301  dev_idx < ec_master_num_devices(master); dev_idx++) {
302  count += fsm->slaves_responding[dev_idx];
303  }
304 
305  if (!count) {
306  // no slaves present -> finish state machine.
307  master->scan_busy = 0;
308  wake_up_interruptible(&master->scan_queue);
309  ec_fsm_master_restart(fsm);
310  return;
311  }
312 
313  size = sizeof(ec_slave_t) * count;
314  if (!(master->slaves =
315  (ec_slave_t *) kmalloc(size, GFP_KERNEL))) {
316  EC_MASTER_ERR(master, "Failed to allocate %u bytes"
317  " of slave memory!\n", size);
318  master->scan_busy = 0;
319  wake_up_interruptible(&master->scan_queue);
320  ec_fsm_master_restart(fsm);
321  return;
322  }
323 
324  // init slaves
325  dev_idx = EC_DEVICE_MAIN;
326  next_dev_slave = fsm->slaves_responding[dev_idx];
327  ring_position = 0;
328  for (i = 0; i < count; i++, ring_position++) {
329  slave = master->slaves + i;
330  while (i >= next_dev_slave) {
331  dev_idx++;
332  next_dev_slave += fsm->slaves_responding[dev_idx];
333  ring_position = 0;
334  }
335 
336  ec_slave_init(slave, master, dev_idx, ring_position, i + 1);
337 
338  // do not force reconfiguration in operation phase to avoid
339  // unnecesssary process data interruptions
340  if (master->phase != EC_OPERATION) {
341  slave->force_config = 1;
342  }
343  }
344  master->slave_count = count;
345  master->fsm_slave = master->slaves;
346 
347  /* start with first device with slaves responding; at least one
348  * has responding slaves, otherwise count would be zero. */
349  fsm->dev_idx = EC_DEVICE_MAIN;
350  while (!fsm->slaves_responding[fsm->dev_idx]) {
351  fsm->dev_idx++;
352  }
353 
354  ec_fsm_master_enter_clear_addresses(fsm);
355  return;
356  }
357  }
358 
359  if (master->slave_count) {
360 
361  // application applied configurations
362  if (master->config_changed) {
363  master->config_changed = 0;
364 
365  EC_MASTER_DBG(master, 1, "Configuration changed.\n");
366 
367  fsm->slave = master->slaves; // begin with first slave
368  ec_fsm_master_enter_write_system_times(fsm);
369 
370  } else {
371  // fetch state from first slave
372  fsm->slave = master->slaves;
373  ec_datagram_fprd(fsm->datagram, fsm->slave->station_address,
374  0x0130, 2);
375  ec_datagram_zero(datagram);
376  fsm->datagram->device_index = fsm->slave->device_index;
377  fsm->retries = EC_FSM_RETRIES;
378  fsm->state = ec_fsm_master_state_read_state;
379  }
380  } else {
381  ec_fsm_master_restart(fsm);
382  }
383 }

391 int ec_fsm_master_action_process_sii(
392  ec_fsm_master_t *fsm
393  )
394 {
395  ec_master_t *master = fsm->master;
396  ec_sii_write_request_t *request;
397 
398  // search the first request to be processed
399  while (1) {
400  if (list_empty(&master->sii_requests))
401  break;
402 
403  // get first request
404  request = list_entry(master->sii_requests.next,
405  ec_sii_write_request_t, list);
406  list_del_init(&request->list); // dequeue
407  request->state = EC_INT_REQUEST_BUSY;
408 
409  // found pending SII write operation. execute it!
410  EC_SLAVE_DBG(request->slave, 1, "Writing SII data...\n");
411  fsm->sii_request = request;
412  fsm->sii_index = 0;
413  ec_fsm_sii_write(&fsm->fsm_sii, request->slave, request->offset,
414  request->words, EC_FSM_SII_USE_CONFIGURED_ADDRESS);
415  fsm->state = ec_fsm_master_state_write_sii;
416  fsm->state(fsm); // execute immediately
417  return 1;
418  }
419 
420  return 0;
421 }

429 int ec_fsm_master_action_process_sdo(
430  ec_fsm_master_t *fsm
431  )
432 {
433  ec_master_t *master = fsm->master;
434  ec_slave_t *slave;
435  ec_sdo_request_t *req;
436 
437  // search for internal requests to be processed
438  for (slave = master->slaves;
439  slave < master->slaves + master->slave_count;
440  slave++) {
441 
442  if (!slave->config) {
443  continue;
444  }
445 
446  list_for_each_entry(req, &slave->config->sdo_requests, list) {
447  if (req->state == EC_INT_REQUEST_QUEUED) {
448 
449  if (ec_sdo_request_timed_out(req)) {
450  req->state = EC_INT_REQUEST_FAILURE;
451  EC_SLAVE_DBG(slave, 1, "Internal SDO request"
452  " timed out.\n");
453  continue;
454  }
455 
456  if (slave->current_state == EC_SLAVE_STATE_INIT) {
457  req->state = EC_INT_REQUEST_FAILURE;
458  continue;
459  }
460 
461  req->state = EC_INT_REQUEST_BUSY;
462  EC_SLAVE_DBG(slave, 1, "Processing internal"
463  " SDO request...\n");
464  fsm->idle = 0;
465  fsm->sdo_request = req;
466  fsm->slave = slave;
467  fsm->state = ec_fsm_master_state_sdo_request;
468  ec_fsm_coe_transfer(&fsm->fsm_coe, slave, req);
469  ec_fsm_coe_exec(&fsm->fsm_coe, fsm->datagram);
470  return 1;
471  }
472  }
473  }
474  return 0;
475 }

483 void ec_fsm_master_action_idle(
484  ec_fsm_master_t *fsm
485  )
486 {
487  ec_master_t *master = fsm->master;
488  ec_slave_t *slave;
489 
490  // Check for pending internal SDO requests
491  if (ec_fsm_master_action_process_sdo(fsm)) {
492  return;
493  }
494 
495  // enable processing of requests
496  for (slave = master->slaves;
497  slave < master->slaves + master->slave_count;
498  slave++) {
499  ec_fsm_slave_set_ready(&slave->fsm);
500  }
501 
502  // check, if slaves have an SDO dictionary to read out.
503  for (slave = master->slaves;
504  slave < master->slaves + master->slave_count;
505  slave++) {
506  if (!(slave->sii.mailbox_protocols & EC_MBOX_COE)
507  || (slave->sii.has_general
508  && !slave->sii.coe_details.enable_sdo_info)
509  || slave->sdo_dictionary_fetched
510  || slave->current_state == EC_SLAVE_STATE_INIT
511  || slave->current_state == EC_SLAVE_STATE_UNKNOWN
512  || jiffies - slave->jiffies_preop < EC_WAIT_SDO_DICT * HZ
513  ) continue;
514 
515  EC_SLAVE_DBG(slave, 1, "Fetching SDO dictionary.\n");
516 
517  slave->sdo_dictionary_fetched = 1;
518 
519  // start fetching SDO dictionary
520  fsm->idle = 0;
521  fsm->slave = slave;
522  fsm->state = ec_fsm_master_state_sdo_dictionary;
523  ec_fsm_coe_dictionary(&fsm->fsm_coe, slave);
524  ec_fsm_coe_exec(&fsm->fsm_coe, fsm->datagram); // execute immediately
525  fsm->datagram->device_index = fsm->slave->device_index;
526  return;
527  }
528 
529  // check for pending SII write operations.
530  if (ec_fsm_master_action_process_sii(fsm)) {
531  return; // SII write request found
532  }
533 
534  ec_fsm_master_restart(fsm);
535 }

569 void ec_fsm_master_action_configure(
570  ec_fsm_master_t *fsm
571  )
572 {
573  ec_master_t *master = fsm->master;
574  ec_slave_t *slave = fsm->slave;
575 
576  if (master->config_changed) {
577  master->config_changed = 0;
578 
579  // abort iterating through slaves,
580  // first compensate DC system time offsets,
581  // then begin configuring at slave 0
582  EC_MASTER_DBG(master, 1, "Configuration changed"
583  " (aborting state check).\n");
584 
585  fsm->slave = master->slaves; // begin with first slave
586  ec_fsm_master_enter_write_system_times(fsm);
587  return;
588  }
589 
590  // Does the slave have to be configured?
591  if ((slave->current_state != slave->requested_state
592  || slave->force_config) && !slave->error_flag) {
593 
594  // Start slave configuration
595  down(&master->config_sem);
596  master->config_busy = 1;
597  up(&master->config_sem);
598 
599  if (master->debug_level) {
600  char old_state[EC_STATE_STRING_SIZE],
601  new_state[EC_STATE_STRING_SIZE];
602  ec_state_string(slave->current_state, old_state, 0);
603  ec_state_string(slave->requested_state, new_state, 0);
604  EC_SLAVE_DBG(slave, 1, "Changing state from %s to %s%s.\n",
605  old_state, new_state,
606  slave->force_config ? " (forced)" : "");
607  }
608 
609  fsm->idle = 0;
610  fsm->state = ec_fsm_master_state_configure_slave;
611  ec_fsm_slave_config_start(&fsm->fsm_slave_config, slave);
612  fsm->state(fsm); // execute immediately
613  fsm->datagram->device_index = fsm->slave->device_index;
614  return;
615  }
616 
617  // process next slave
618  ec_fsm_master_action_next_slave_state(fsm);
619 }

627 void ec_fsm_master_state_read_state(
628  ec_fsm_master_t *fsm
629  )
630 {
631  ec_slave_t *slave = fsm->slave;
632  ec_datagram_t *datagram = fsm->datagram;
633 
634  if (datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
635  return;
636  }
637 
638  if (datagram->state != EC_DATAGRAM_RECEIVED) {
639  EC_SLAVE_ERR(slave, "Failed to receive AL state datagram: ");
640  ec_datagram_print_state(datagram);
641  ec_fsm_master_restart(fsm);
642  return;
643  }
644 
645  // did the slave not respond to its station address?
646  if (datagram->working_counter != 1) {
647  if (!slave->error_flag) {
648  slave->error_flag = 1;
649  EC_SLAVE_DBG(slave, 1, "Slave did not respond to state query.\n");
650  }
651  fsm->rescan_required = 1;
652  ec_fsm_master_restart(fsm);
653  return;
654  }
655 
656  // A single slave responded
657  ec_slave_set_state(slave, EC_READ_U8(datagram->data));
658 
659  if (!slave->error_flag) {
660  // Check, if new slave state has to be acknowledged
661  if (slave->current_state & EC_SLAVE_STATE_ACK_ERR) {
662  fsm->idle = 0;
663  fsm->state = ec_fsm_master_state_acknowledge;
664  ec_fsm_change_ack(&fsm->fsm_change, slave);
665  fsm->state(fsm); // execute immediately
666  return;
667  }
668 
669  // No acknowlegde necessary; check for configuration
670  ec_fsm_master_action_configure(fsm);
671  return;
672  }
673 
674  // slave has error flag set; process next one
675  ec_fsm_master_action_next_slave_state(fsm);
676 }

965 u64 ec_fsm_master_dc_offset32(
966  ec_fsm_master_t *fsm,
967  u64 system_time,
968  u64 old_offset,
969  unsigned long jiffies_since_read
970  )

975

978

982  time_diff = (u32) slave->master->app_time - system_time32;

988  slave->master->app_time, time_diff);
989 
990  if (EC_ABS(time_diff) > EC_SYSTEM_TIME_TOLERANCE_NS) {
991  new_offset = time_diff + old_offset32;
992  EC_SLAVE_DBG(slave, 1, "Setting time offset to %u (was %u)\n",
993  new_offset, old_offset32);
994  return (u64) new_offset;
995  } else {
996  EC_SLAVE_DBG(slave, 1, "Not touching time offset.\n");
997  return old_offset;
